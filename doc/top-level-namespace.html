post '/bokning/:id/betyg' do |id|
  bokning_id = id.to_i

  redirect '/konto' unless auth?
  redirect '/konto' unless user_can_rate(bokning_id, session[:user])

  text = params[:text]
  betyg = params[:rating].to_i
  user_id = session[:user][:id]
  rate_room(bokning_id, user_id, text, betyg)
  redirect '/konto'
end
